```typescript
export type RunStatus = "pending" | "running" | "completed" | "failed" | "cancelled";
export type JobStatus = "pending" | "running" | "success" | "failed" | "skipped";
```

Add `packages/types/src/paths.ts` with these deterministic path utilities from `spec/07-storage.md`:

```typescript
export function runLogPath(namespaceId: string, runId: string, jobId: string): string {
  return `${namespaceId}/runs/${runId}/logs/${jobId}.log`;
}

export function planPath(namespaceId: string, checksum: string): string {
  return `${namespaceId}/plans/${checksum}.json`;
}

export function coordinatorKey(namespaceId: string, runId: string): string {
  return `${namespaceId}:${runId}`;
}
```

Update `packages/types/package.json` so consumers can import:

```typescript
import type { Run, Job } from "@orun/types";
import { runLogPath, planPath, coordinatorKey } from "@orun/types/paths";
```

Keep the existing scaffold build workflow working. If you change package exports, ensure both source-level TypeScript resolution and emitted `dist` builds still work.

Add focused tests for `packages/types/src/paths.ts`:

- `runLogPath("123", "run-1", "job-a")` returns `123/runs/run-1/logs/job-a.log`
- `planPath("123", "abc")` returns `123/plans/abc.json`
- `coordinatorKey("123", "run-1")` returns `123:run-1`

Add type-level coverage where practical. Keep it lightweight; do not introduce heavy compile-time test frameworks. A small `.test.ts` file that imports the public API and uses representative values is enough.
